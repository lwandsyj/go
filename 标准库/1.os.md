1. os.Args main 函数不接受任何参数和返回值，使用os.Args 接收传入的参数

        package main

        import (
            "fmt"
            "os"
        )

        func main(){
            // 获取参数，第一个参数为当前项目生成的二进制文件
            //[/var/folders/4j/fh0m772n463bq3h7dnxtfhgh0000gn/T/go-build686622546/b001/exe/main test]
            // 从第二个参数开始是正式传入的参数
            a:=os.Args
            fmt.Println(a)
            // 截取参数
            args:=a[1:]
            fmt.Println(args)
        }

        go run main.go test(传入的参数)
